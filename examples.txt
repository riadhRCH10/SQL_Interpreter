***** CREATE *****

CREATe TABLE PERSONNE (
    nom VARCHAR,
    prenom VARCHAR,
    id int primary key
)

CREATE TABLE orders (
    order_id INTEGER PRIMARY KEY,
    customer_id INTEGER,
    order_date DATE,
    FOREIGN KEY customer_id REFERENCES customers(customer_id)   //constraints
);

CREATE TABLE products (
    product_id INTEGER PRIMARY KEY,
    product_name VARCHAR(50),
    price DECIMAL(10,2),
    UNIQUE (product_name)
);
